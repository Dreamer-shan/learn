```java
public OrderResult getOrderResult(OrderCondition condition, ListenableFuture<OrderCheckResult> checkDupOrderFuture) {
        long start = System.currentTimeMillis();
        String flightOrderNo = null;
        // “清理订单”标志
        boolean needClearOrder = true;
        OrderResult orderResult = null;
        try {
            // 已经生成成功的子订单map<类型，订单号集合>
            Map<OrderType, SubOrderCreateResult> orderCreateResultMap =
                    Collections.synchronizedMap(new HashMap<OrderType, SubOrderCreateResult>());

            // 1.机票生单之前，先取出后续生单所需的BookingTag信息
            OrderReq orderReq = condition.getOrderReq();

            BookingTagInfoForPackage flightBookingTagInfo;
            SubOrderCreateRequest orderCreateRequest;
            if (commonQConfig.getBoolean(QConfigKeys.BooleanKeys.CREATE_ORDER_BUILD_REQUEST_NEW_SWITCH)) {
                //重置bookingTag,获取生子单所需的bookingTagInfoForPachage
                Pair<BookingTagInfoForPackage, SubOrderCreateRequest> pair = buildSubOrderCreateRequest(condition, orderCreateResultMap);

                flightBookingTagInfo = pair.getFirst();
                if (flightBookingTagInfo == null) {
                    QMonitor.recordOne(QMonitorKey.ORDER_CREATE_BOOKINGTAG_GET_ERROR.key);
                    return orderResult = OrderControllerUtil.defaultErrorResult();
                } else if (!flightBookingTagInfo.isBookingTagExists()) {
                    QMonitor.recordOne(QMonitorKey.ORDER_CREATE_BOOKINGTAG_LOST.key);
                    logger.error("createOrder getOrderResult isBookingTagExists=false");
                    return orderResult = OrderControllerUtil.bookingTagLostError();
                } else if (flightBookingTagInfo.isBookingAbsIntercept()) {
                    QMonitor.recordOne(QMonitorKey.ORDER_CREATE_BOOKINGTAG_ABSINTERCEPT.key);
                    logger.error("createOrder getOrderResult isBookingAbsIntercept=true");
                    return orderResult = OrderControllerUtil.bookingAbsInterceptError(condition);
                }

                orderCreateRequest = pair.getSecond();
            } else {
                flightBookingTagInfo = fetchBookingTagForPackage(orderReq);
                // 根据不同的返回结果，返回不同的状态码
                if (flightBookingTagInfo == null) {
                    QMonitor.recordOne(QMonitorKey.ORDER_CREATE_BOOKINGTAG_GET_ERROR.key);
                    return orderResult = OrderControllerUtil.defaultErrorResult();
                } else if (!flightBookingTagInfo.isBookingTagExists()) {
                    QMonitor.recordOne(QMonitorKey.ORDER_CREATE_BOOKINGTAG_LOST.key);
                    return orderResult = OrderControllerUtil.bookingTagLostError();
                } else if (flightBookingTagInfo.isBookingAbsIntercept()) {
                    return orderResult = OrderControllerUtil.bookingAbsInterceptError(condition);
                }

                AppContext.setOtaSite(orderReq.getClientSite());

                // 2.构造生单请求
                orderCreateRequest = resetReqAndBuildCreateOrderParam(condition, orderReq, flightBookingTagInfo, orderCreateResultMap);
            }

            if (orderCreateRequest == null) {
                logger.error("构造生单请求失败");
                return orderResult = OrderControllerUtil.defaultErrorResult();
            }

            // 3. 判断重复订单
            OrderCheckResult dupOrderCheckResult = dupOrderCheckService.getDupOrderCheckResult(checkDupOrderFuture);
            if (dupOrderCheckResult != null) {
                QMonitor.recordOne("dupOrderCheck_ret_" + dupOrderCheckResult.isOrderDup());
                if (dupOrderCheckResult.isOrderDup()) {
                    logger.info("重复订单判断结果不通过，弹框提示");
                    orderResult = new OrderResult();
                    orderResult.setCode(OrderWirelessResult.ERROR_TYPE.five_hundred_eight.CODE);
                    orderResult.setOrderState(dupOrderCheckResult.getStatus());
                    orderResult.setMsg(dupOrderCheckResult.getErrMsg());
                    return orderResult;
                } else {
                    logger.info("重复订单判断结果通过，继续生单");
                }
            } else {
                QMonitor.recordOne("dupOrderCheck_ret_null");
                logger.info("未获取到重复订单判断结果，继续生单");
            }

            // 补充机票订单信息
            flightOrderNo = generateFlightOrderNo(orderReq);
            AppContext.setOrderNoForLog(flightOrderNo);


            FlightInfoForSubOrder flightOrderInfo = buildFlightOrderInfo(flightOrderNo, orderReq);
            if (flightOrderInfo == null) {
                QMonitor.recordOne("createFlightOrder_queryFlightOrderInfo_fail");
                logger.error("生机票订单后查询失败，结果为空");
                return orderResult = OrderControllerUtil.defaultErrorResult();
            }
            orderCreateRequest.setFlightOrderInfo(flightOrderInfo);

            // 4.生单

            // 校验营销刷新风险
            if (!activityCheckService.check(orderReq, flightBookingTagInfo)) {
                QMonitor.recordOne("checkActivityOnCreateOrder_fail");
                logger.warn("营销校验失败，存在刷新风险，返回失败");
                String tipMsg = commonQConfig.getHotConfig("activity_check_false_msg");
                tipMsg = StringUtils.defaultIfEmpty(tipMsg, DEFAULT_ACTIVITY_CHECK_MSG);
                return orderResult = OrderControllerUtil.getErrorOrderResult(404, tipMsg, null, null);
            }

            // 计算有哪些子订单需要创建（包括机票单）
            Set<OrderType> subOrderSet = computeSubOrderList(orderCreateRequest);
            logger.info("开始生子单，子单列表:{}", subOrderSet);
            // 完成生单的子订单集合
            Set<OrderType> finishedSet = Collections.synchronizedSet(new HashSet<OrderType>());
            // 生子单
            createSubOrders(subOrderSet, flightOrderNo, orderCreateRequest, finishedSet, orderCreateResultMap);
            SubOrderCreateResult flightOrderResult = orderCreateResultMap.get(OrderType.N_FLIGHT);
            if (flightOrderResult == null) {
                QMonitor.recordOne("createFlightOrder_fail");
                logger.error("生成机票订单失败，结果为空");
                return orderResult = OrderControllerUtil.defaultErrorResult();
            }

            orderResult = OrderControllerUtil.convertFlightOrderResult(flightOrderResult.getFlightOrderCreateResult());
            logger.info("生单结果flightOrderResult转换为orderResult= {}", JSON.toJSONString(orderResult));
            // 机票生单失败，则马上返回，告知用户生单失败
            if (!orderResult.isSuccess()) {
                QMonitor.recordOne("OLD_CREATE_ORDER_FAIL");
                PackageMonitorUtil.monitorFlightOrderCreateResult(flightOrderResult.getFlightOrderCreateResult());
                //www生单失败，拦截进NTS
                return orderResult;
            }
            //判断是否需要阻断的子单 需要阻断 告知用户生单失败
            SubOrderCreateResult promotionOrderResult =  orderCreateResultMap.get(OrderType.PROMOTION);
            if(promotionOrderResult!=null && promotionOrderResult.getResultType() == OrderCreateResultType.BLOCK) {
                //清除已生成机票单标识
                needClearOrder = true;
                String msg = promotionOrderResult.getDirectCutPromotionResult().getMessgage();
                QMonitor.recordOne("SUB_ORDER_BLOCK_ORDER_PROMOTION");
                orderResult =  OrderControllerUtil.defaultErrorResult();
                orderResult.setCode(OrderWirelessResult.ERROR_TYPE.five_hundred_nine.CODE);
                Map<String, Object> resultToView = new HashMap<String, Object>();
                resultToView.put(OrderResultToViewKeyEnum.SUBORDERERRORMSG.key,msg);
                orderResult.setOrderResultToView(resultToView);
                orderResult.setMsg(msg);
                return orderResult;
            }

            //判断代金券是否需要阻断 需要阻断 告知用户生单失败
            SubOrderCreateResult bonusResult =  orderCreateResultMap.get(OrderType.BONUS);
            if(bonusResult != null && bonusResult.getResultType() == OrderCreateResultType.BLOCK) {
                if (commonQConfig.getBoolean(QConfigKeys.BooleanKeys.SPECIAL_PRODUCT_MARK_MUST_HAVE_BONUS_BLOCK_SWITCH)) {
                    //清除已生成机票单标识
                    needClearOrder = true;
                    String msg = bonusResult.getBonusCreateOrderResult().getMsg();
                    QMonitor.recordOne("SUB_ORDER_BLOCK_ORDER_BONUS");
                    orderResult =  OrderControllerUtil.defaultErrorResult();
                    orderResult.setCode(OrderWirelessResult.ERROR_TYPE.five_hundred_nine.CODE);
                    Map<String, Object> resultToView = new HashMap<String, Object>();
                    resultToView.put(OrderResultToViewKeyEnum.SUBORDERERRORMSG.key,msg);
                    orderResult.setOrderResultToView(resultToView);
                    orderResult.setMsg(msg);
                    return orderResult;
                }
            }

            flightOrderResult.setPayAmount(BigDecimal.valueOf(flightOrderInfo.getOrderInfo().getAllPrices()));
            // 更改机票订单信息，兼容售后交易
            boolean fixResult = fixFlightOrderInfo(orderResult, orderCreateResultMap, orderReq, flightBookingTagInfo);

            long start2 = System.currentTimeMillis();
            // 计算失败子订单集合
            try {
                Set<OrderType> errorList = computeErrorList(subOrderSet, finishedSet, fixResult);
                logger.info("getOrderResult failureSubOrder:{}", errorList);

                // 检查失败子订单是否是允许的否则返回生单失败
                if (!checkSubOrderErrorCanBeTolerated(errorList)) {
                    logger.error("getOrderResult failureSubOrder error:{}", errorList);
                    if (orderResult.getOrderResultToView() != null && "1".equals(orderResult.getOrderResultToView().get("interceptOrder"))) {
                        return orderResult;
                    }
                    orderResult = OrderControllerUtil.defaultErrorResult();
                    SubOrderCreateResult flightHotelOrderResult = orderCreateRequest.getNonSuccessCreateResultMap().get(OrderType.HOTEL);
                    if (Objects.nonNull(flightHotelOrderResult) && flightHotelOrderResult.getResultType() == OrderCreateResultType.FAILURE) {
                        buildMassage(orderResult, flightHotelOrderResult);
                    }
                    return orderResult;
                }

                // 子单失败,不影响整单失败监控
                if (CollectionUtils.isNotEmpty(errorList)){
                    logger.info("子单失败不影响整单失败:{}", flightOrderNo);
                    QMonitor.recordOne(QMonitorKey.SUB_ORDER_FAIL_NOT_CAUSE_ALL_CANCEL.key);
                }

                // 生单成功，设置“清理订单”为false
                needClearOrder = false;

                // 合并 子订单生单结果到 返给用户的参数里
                mergeSubOrderCreateInfoToResult(subOrderSet, orderResult, orderCreateRequest, errorList, !QStringUtils.isEmptyJsonArray(orderReq.getCashCoupon()));
                // 统计特定生单成功订单
                staticOrderLog(flightOrderInfo);
                sendAsynOrderMqService.sendAsynOrderService(flightOrderNo);
            } finally {
                QMonitor.recordQuantile("after_createOrder_deal_tips", System.currentTimeMillis() - start2);
            }
            if (qlibraService.isFlightTypeNeedReport(orderReq.getFlightType())) {
                qlibraService.reportTransferPackageXiAn(flightBookingTagInfo, orderResult);
            }

            if (orderResult.isSuccess()) {
                activityCheckService.record(orderReq, flightBookingTagInfo);
            }

            return orderResult;
        } catch (Exception e) {
            logger.error("getOrderResult error", e);
            QMonitor.recordOne(QMonitorKey.TOTAL_ORDER_CREATE_EXCEPTION.key);
            return orderResult = OrderControllerUtil.defaultErrorResult();
        } finally {
            if(StringUtils.isBlank(orderResult.getOrderNo()) && StringUtils.isNotBlank(flightOrderNo)){
                orderResult.setOrderNo(flightOrderNo);
            }
//            logger.info("create order total cost time : {} ms", System.currentTimeMillis() - start);
            QMonitor.recordOne(condition.getBookingType() + QMonitorKey.ORDER_CREATE.key + condition.getCaller(),
                    System.currentTimeMillis() - start);
            PackageMonitorUtil.recordOne(MonitorType.TOTAL_ORDER.getCode(), condition.getOrderReq(), System.currentTimeMillis() - start);
            QMonitor.recordQuantile("webPackageOrderService_getOrderResult_total", System.currentTimeMillis() - start);
            boolean isSuccess = orderResult != null && orderResult.isSuccess();
            tradeNotifyMsgService.sendCreateOrderMessage(condition, isSuccess ? flightOrderNo : "", System.currentTimeMillis() - start);

            if (needClearOrder && StringUtils.isNotBlank(flightOrderNo)) {
                if (QProductConfig.getFlagShipDomainSet().contains(AppContext.getOtaSite())) {
                    QMonitor.recordOne(QMonitorKey.AFARE_FLAGSHIP_ORDER_CANCEL.key + CommonUtil.getPrefixOfDomain(AppContext.getOtaSite()));
                }
                // 当因为子订单失败造成生单失败时，取消不完整的机票订单
                clearImperfectOrder(flightOrderNo);
            }
        }
    }
```

```java
private Pair<BookingTagInfoForPackage, SubOrderCreateRequest> buildSubOrderCreateRequest(OrderCondition condition, Map<OrderType, SubOrderCreateResult> orderCreateResultMap) {
        //生单输入参数
        OrderReq orderReq = condition.getOrderReq();

        Stopwatch stopwatch = Stopwatch.createStarted();
        try {
            int timeout = commonQConfig.getInt(QConfigKeys.IntKeys.CREATE_ORDER_FETCH_BOOKINGTAG_CALLABLE_TIMEOUT);
            //syncBooking为true代表同步
            //异步
            ListenableFuture<Pair<BookingTagInfoForPackage, PriceInfoForOrderCreate>> asynFuture =
                    commonCombineExecutorService.submit(new CreateOrderPreHandler(false, orderReq, bookingTagManageService));
            //同步
            ListenableFuture<Pair<BookingTagInfoForPackage, PriceInfoForOrderCreate>> syncFuture =
                    commonCombineExecutorService.submit(new CreateOrderPreHandler(true, orderReq, bookingTagManageService));
            //获取异步结果
            Pair<BookingTagInfoForPackage, PriceInfoForOrderCreate> asynPair = asynFuture.get(timeout, TimeUnit.MILLISECONDS);

            if (asynPair == null) {
                QMonitor.recordOne("get_asynPair_from_ListenableFuture_null");
                logger.error("get asynPair from ListenableFuture null");
                return new Pair<>(null, null);
            }

            BookingTagInfoForPackage asynBookingTagInfoForPackage = asynPair.getFirst();
            if (asynBookingTagInfoForPackage == null) {
                QMonitor.recordOne("get_asynBookingTagInfoForPackage_from_ListenableFuture_null");
                logger.error("get asynBookingTagInfoForPackage from ListenableFuture null");
                return new Pair<>(null, null);
            } else if (!asynBookingTagInfoForPackage.isBookingTagExists()
                    || asynBookingTagInfoForPackage.isBookingAbsIntercept()) {
                return new Pair<>(asynBookingTagInfoForPackage, null);
            }

            PriceInfoForOrderCreate priceInfoForOrderCreate = asynPair.getSecond();
            if (priceInfoForOrderCreate == null) {
                QMonitor.recordOne(QMonitorKey.ORDER_CREATE_BUILD_PRICEINFO_ERROR.key);
                logger.error("get priceInfoForOrderCreate from ListenableFuture null");
                return new Pair<>(asynBookingTagInfoForPackage, null);
            }

            // 重置生单信息
            resetSubProducts(orderReq, asynBookingTagInfoForPackage, priceInfoForOrderCreate);

            // 构建生单所需参数
            SubOrderCreateRequest subOrderCreateRequest = buildOrderCreateRequestParam(condition, asynBookingTagInfoForPackage, priceInfoForOrderCreate, orderCreateResultMap);

            // 等同步bookingTag处理完成
            Pair<BookingTagInfoForPackage, PriceInfoForOrderCreate> syncPair = syncFuture.get(timeout, TimeUnit.MILLISECONDS);
            if (syncPair == null) {
                QMonitor.recordOne("get_syncPair_from_ListenableFuture_null");
                logger.error("get syncPair from ListenableFuture null");
                return new Pair<>(null, null);
            }
            // 正常返回
            return new Pair<>(asynBookingTagInfoForPackage, subOrderCreateRequest);
        } catch (InterruptedException e) {
            logger.error("buildSubOrderCreateRequest InterruptedException:", e);
            QMonitor.recordOne("buildSubOrderCreateRequest_InterruptedException");
        } catch (ExecutionException e) {
            logger.error("buildSubOrderCreateRequest ExecutionException:", e);
            QMonitor.recordOne("buildSubOrderCreateRequest_ExecutionException");
        } catch (TimeoutException e) {
            logger.error("buildSubOrderCreateRequest TimeoutException:", e);
            QMonitor.recordOne("buildSubOrderCreateRequest_TimeoutException");
        } catch (Exception e) {
            logger.error("buildSubOrderCreateRequest Exception:", e);
            QMonitor.recordOne("buildSubOrderCreateRequest_Exception");
        } finally {
            QMonitor.recordQuantile("buildSubOrderCreateRequest_total", stopwatch.elapsed(TimeUnit.MILLISECONDS));
        }

        return new Pair<>(null, null);
    }
```
